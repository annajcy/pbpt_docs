117 // Handle negative input, which is invalid for real square roots.
118 if (x < 0) {
119 if (std::is_constant_evaluated()) {
120 // Triggers a descriptive compile-time error.
121 throw "Compile-time error: sqrt of negative number";
122 } else {
123 // Throws a standard exception at runtime.
124 // Returning std::sqrt(x) would produce a quiet NaN, which can be harder to debug.
125 throw std::runtime_error("sqrt of negative number");
