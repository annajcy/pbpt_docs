54 std::vector<std::pair<int, std::vector<T>>> load_data() {
55 std::ifstream fin(file_path_);

57
58 std::vector<std::pair<int, std::vector<T>>> data;

61 while (std::getline(fin, line)) {
62 if (should_skip_line(line)) continue;

64 auto parsed = parse_line(line);

79 std::pair<int, std::vector<T>> parse_line(const std::string& line) const {
80 std::string processed_line = line;
81 std::replace(processed_line.begin(), processed_line.end(), delimiter_, ' ');
82
83 std::istringstream iss(processed_line);
84 int wavelength;
85 if (!(iss >> wavelength)) return {-1, {}};
86
87 std::vector<T> values;
88 T value;
89 while (iss >> value) {
90 values.push_back(value);
91 }
92
93 return {wavelength, values};
94 }
