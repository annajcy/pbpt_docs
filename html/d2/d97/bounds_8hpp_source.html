243template <typename T, int N>
244std::optional<std::pair<T, T>> intersect_ray_bounds(
245 const Ray<T, N>& ray,
246 const Bounds<T, N>& bounds,
247 std::pair<T, T> t_range = {
248 -std::numeric_limits<T>::infinity(),
249 std::numeric_limits<T>::infinity()
250 }
251) {
252 T t_min = t_range.first;
253 T t_max = t_range.second;
254
255 const auto& o = ray.origin();
256 const auto& d = ray.direction();
257
258 for (int i = 0; i < N; ++i) {
259 if (is_zero(d[i])) {
260 // 平行该轴方向的 slab
261 if (o[i] < bounds.min()[i] || o[i] > bounds.max()[i])
262 return std::nullopt;
263 } else {
264 T inv_d = T(1) / d[i];
265 T t0 = (bounds.min()[i] - o[i]) * inv_d;
266 T t1 = (bounds.max()[i] - o[i]) * inv_d;
267
268 if (t0 > t1) std::swap(t0, t1);
269
270 t_min = std::max(t_min, t0);
271 t_max = std::min(t_max, t1);
272
273 if (t_min > t_max)
274 return std::nullopt;
275 }
276 }
277
278 return std::pair{t_min, t_max};
279}
