30template<typename FilmType>
31void write_exr(
32 const FilmType& film,
33 const std::filesystem::path& output_path,
34 int width,
35 int height
36) {
37 if (!output_path.parent_path().empty()) {
38 std::filesystem::create_directories(output_path.parent_path());
39 }
40
41 std::vector<float> buffer(
42 static_cast<std::size_t>(width) * static_cast<std::size_t>(height) * 3
43 );
44
45 for (int y = 0; y < height; ++y) {
46 for (int x = 0; x < width; ++x) {
47 auto rgb = film.get_pixel_rgb(math::Point<int, 2>(x, y));
48 std::size_t idx = (
49 static_cast<std::size_t>(height - 1 - y) * static_cast<std::size_t>(width) +
50 static_cast<std::size_t>(x)
51 ) * 3;
52 buffer[idx + 0] = static_cast<float>(rgb.r());
53 buffer[idx + 1] = static_cast<float>(rgb.g());
54 buffer[idx + 2] = static_cast<float>(rgb.b());
55 }
56 }
57
58 const std::size_t pixel_stride = sizeof(float) * 3;
59 const std::size_t row_stride = pixel_stride * static_cast<std::size_t>(width);
60 auto* base = reinterpret_cast<char*>(buffer.data());
61
62 Imf::Header header(width, height);
63 header.channels().insert("R", Imf::Channel(Imf::FLOAT));
64 header.channels().insert("G", Imf::Channel(Imf::FLOAT));
65 header.channels().insert("B", Imf::Channel(Imf::FLOAT));
66
67 Imf::FrameBuffer frame_buffer;
68 frame_buffer.insert("R", Imf::Slice(Imf::FLOAT, base, pixel_stride, row_stride));
69 frame_buffer.insert("G", Imf::Slice(Imf::FLOAT, base + sizeof(float), pixel_stride, row_stride));
70 frame_buffer.insert("B", Imf::Slice(Imf::FLOAT, base + sizeof(float) * 2, pixel_stride, row_stride));
71
72 try {
73 Imf::OutputFile file(output_path.string().c_str(), header);
74 file.setFrameBuffer(frame_buffer);
75 file.writePixels(height);
76 } catch (const std::exception& e) {
77 throw std::runtime_error("Failed to write image to " + output_path.string() + ": " + e.what());
78 }
79}
