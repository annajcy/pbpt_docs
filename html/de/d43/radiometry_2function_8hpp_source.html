18template<typename T, int N>
19SampledWavelength<T, N> create_wavelength_samples_uniform() {
20 math::Vector<T, N> values{};
21 const T min_lambda = radiometry::lambda_min<T>;
22 const T max_lambda = radiometry::lambda_max<T>;
23 const T span = max_lambda - min_lambda;
24 const T step = span / static_cast<T>(N);
25 for (int i = 0; i < N; ++i) {
26 values[i] = min_lambda + step * (static_cast<T>(i) + T(0.5));
27 }
28 return SampledWavelength<T, N>(values);
29}
