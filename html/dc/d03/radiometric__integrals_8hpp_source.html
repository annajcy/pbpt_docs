201 ParallelogramAreaDomain(const math::Point<T, 3>& origin, const math::Vector<T, 3>& edge1, const math::Vector<T, 3>& edge2)
202 : m_origin(origin), m_edge1(edge1), m_edge2(edge2) {

251template<typename T, typename Domain, typename Func, typename RNG>
252auto integrate(const Domain& domain, const Func& func, int sample_count, RNG& rng) {
253 using R = decltype(std::declval<const Func&>()(std::declval<typename Domain::sample_type>()) / static_cast<T>(std::declval<int>()));
254 R result = R{};
255 domain.foreach_sample(sample_count, rng, [&](const auto& sample, const auto& pdf) {
256 result += func(sample) / pdf;
257 });
258 return result / T(sample_count);
259}
