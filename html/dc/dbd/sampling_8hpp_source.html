201template<typename T>
202inline math::Point<T, 2> sample_uniform_disk_concentric(const math::Point<T, 2>& uv, T radius = 1.0) {
203 math::Point<T, 2> p_offset = 2.0 * uv.to_vector() - math::Vector<T, 2>(1, 1);
204
205 if (p_offset.x() == 0 && p_offset.y() == 0) {
206 return math::Point<T, 2>(0, 0);
207 }
208
209 T theta{}, r{};
210 if (std::abs(p_offset.x()) > std::abs(p_offset.y())) {
211 r = p_offset.x();
212 theta = (math::pi_v<T> / 4.0) * (p_offset.y() / p_offset.x());
213 } else {
214 r = p_offset.y();
215 theta = (math::pi_v<T> / 2.0) - (math::pi_v<T> / 4.0) * (p_offset.x() / p_offset.y());
216 }
217
218 return math::Point<T, 2>(radius * r * math::Vector<T, 2>(std::cos(theta), std::sin(theta)));
219}

245 T z = uv.x();
