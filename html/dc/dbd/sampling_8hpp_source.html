197template<typename T>
198inline math::Point<T, 2> sample_uniform_disk_concentric(const math::Point<T, 2>& uv, T radius = 1.0) {
199 math::Point<T, 2> p_offset = 2.0 * uv.to_vector() - math::Vector<T, 2>(1, 1);
200
201 if (p_offset.x() == 0 && p_offset.y() == 0) {
202 return math::Point<T, 2>(0, 0);
203 }
204
205 T theta{}, r{};
206 if (std::abs(p_offset.x()) > std::abs(p_offset.y())) {
207 r = p_offset.x();
208 theta = (math::pi_v<T> / 4.0) * (p_offset.y() / p_offset.x());
209 } else {
210 r = p_offset.y();
211 theta = (math::pi_v<T> / 2.0) - (math::pi_v<T> / 4.0) * (p_offset.x() / p_offset.y());
212 }
213
214 return math::Point<T, 2>(radius * r * math::Vector<T, 2>(std::cos(theta), std::sin(theta)));
215}
