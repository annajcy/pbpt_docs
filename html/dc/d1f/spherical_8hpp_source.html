27template<typename T>
28inline constexpr T wrap_angle_2pi(T phi) {
29 if (is_less(phi, T(0))) phi += T(2) * pi_v<T>;
30 else if (is_greater_equal(phi, T(2) * pi_v<T>)) phi -= T(2) * pi_v<T>;
31 else return phi;
32 return wrap_angle_2pi(phi);
33}

51class SphericalPoint {
52public:
54 Vector<T, N - 1> m_spherical;
56 T m_radius;

64 constexpr SphericalPoint(const Vector<T, N - 1>& spherical, T radius) : m_spherical(spherical), m_radius(radius) {}
66 constexpr SphericalPoint(const Point<T, N>& cartesian) {
67 m_radius = cartesian.to_vector().length();
68
69 if (is_zero(m_radius)) {
70 m_spherical = Vector<T, N - 1>::zeros();
71 return;
72 }
73
74 if constexpr (N == 2) {
75 // 2D情况: φ = atan2(y, x) (方位角)
76 m_spherical[0] = wrap_angle_2pi(atan2(cartesian.y(), cartesian.x()));
77 } else if constexpr (N == 3) {
78 // 3D情况: θ = acos(z/r), φ = atan2(y, x)
79 m_spherical[0] = wrap_angle_2pi(acos(cartesian.z() / m_radius)); // 极角
80 m_spherical[1] = wrap_angle_2pi(atan2(cartesian.y(), cartesian.x())); // 方位角
81 } else {
82 // N维情况的递归计算
83 cartesian_to_spherical_nd(cartesian);
84 }
85 }
86
88 constexpr Point<T, N> to_cartesian() const {
89 if (m_radius == 0) {
90 return Point<T, N>::zeros();
91 }
92
93 Vector<T, N> result;
94
95 if constexpr (N == 2) {
96 // 2D: x = r*cos(φ), y = r*sin(φ)
97 result[0] = m_radius * cos(m_spherical[0]);
98 result[1] = m_radius * sin(m_spherical[0]);
99 } else if constexpr (N == 3) {
100 // 3D: x = r*sin(θ)*cos(φ), y = r*sin(θ)*sin(φ), z = r*cos(θ)
101 T sin_theta = sin(m_spherical[0]);
102 result[0] = m_radius * sin_theta * cos(m_spherical[1]);
103 result[1] = m_radius * sin_theta * sin(m_spherical[1]);
104 result[2] = m_radius * cos(m_spherical[0]);
105 } else {
106 // N维情况
107 spherical_to_cartesian_nd(result);
108 }
109
110 return Point<T, N>::from_vector(result);
111 }
112
114 constexpr T radius() const { return m_radius; }
116 constexpr T angle(int i) const { return m_spherical[i]; }
118 constexpr const Vector<T, N - 1>& angles() const { return m_spherical; }

121 constexpr T azimuth() const {
122 static_assert(N >= 2, "Azimuth angle only exists for N >= 2");

124 }

128 constexpr void cartesian_to_spherical_nd(const Point<T, N>& cartesian) {
129 // N维球坐标的一般公式（方位角在最后）
130 // θᵢ = acos(xᵢ₊₂ / sqrt(x₁² + x₂² + ... + xᵢ₊₂²)) for i = 0, ..., N-3
131 // (极角) φ = atan2(x₂, x₁) (方位角)
132
133 // 计算极角 (前N-2个角度)
134 for (int i = 0; i < N - 2; ++i) {
135 T sum_squares = 0;
136 for (int j = 0; j <= i + 2; ++j) {
137 sum_squares += cartesian[j] * cartesian[j];
138 }
139 T partial_radius = sqrt(sum_squares);
140
141 if (is_zero(partial_radius)) {
142 m_spherical[i] = 0;
143 } else {
144 m_spherical[i] = acos(cartesian[i + 2] / partial_radius);
145 }
146 }
147
148 // 最后一个角度 (方位角)
149 m_spherical[N - 2] = wrap_angle_2pi(atan2(cartesian[1], cartesian[0]));
150 }
151
152 constexpr void spherical_to_cartesian_nd(Vector<T, N>& result) const {
153 // N维球坐标到笛卡尔的一般公式（方位角在最后）
154 // x₁ = r * cos(φ) * ∏ᵢ₌₀ᴺ⁻³ sin(θᵢ)
155 // x₂ = r * sin(φ) * ∏ᵢ₌₀ᴺ⁻³ sin(θᵢ)
156 // xₖ = r * cos(θₖ₋₂) * ∏ᵢ₌ₖ₋₂ᴺ⁻³ sin(θᵢ) for k = 3, ..., N-1
157 // xₙ = r * cos(θₙ₋₃)
158
159 // 计算所有sin值的累积乘积
160 Vector<T, N - 1> sin_products;
161 sin_products[N - 2] = T(1); // 最后一个元素（方位角位置）
162
163 for (int i = N - 3; i >= 0; --i) {
164 sin_products[i] = sin_products[i + 1] * sin(m_spherical[i]);
165 }
166
167 // 计算笛卡尔坐标
168 T azimuth_angle = m_spherical[N - 2]; // 方位角
169 result[0] = m_radius * cos(azimuth_angle) * sin_products[0];
170 result[1] = m_radius * sin(azimuth_angle) * sin_products[0];
171
172 for (int i = 2; i < N; ++i) {
173 if (i == N - 1) {
174 // 最后一个坐标
175 result[i] = m_radius * cos(m_spherical[i - 2]);
176 } else {
177 result[i] = m_radius * cos(m_spherical[i - 2]) * sin_products[i - 1];
178 }
179 }
180 }
181};

241template<typename T>
242inline constexpr auto cos_delta_phi(const Vector<T, 3>& a, const Vector<T, 3>& b) {
243 auto axy = a.x() * a.x() + a.y() * a.y();
244 auto bxy = b.x() * b.x() + b.y() * b.y();
245 return (is_zero(axy) || is_zero(bxy)) ? T(1) : std::clamp(
246 (a.x() * b.x() + a.y() * b.y()) / std::sqrt(axy * bxy),
247 T(-1), T(1)
248 );
249}

257template<typename T>
258inline constexpr auto warp_equal_area_square(math::Point<T, 2> uv) {
259 if (uv.x() < 0) {
260 uv.x() = -uv.x(); // mirror across u = 0
261 uv.y() = 1 - uv.y(); // mirror across v = 0.5
262 } else if (uv.x() > 1) {
263 uv.x() = 2 - uv.x(); // mirror across u = 1
264 uv.y() = 1 - uv.y(); // mirror across v = 0.5
265 }
266 if (uv.y() < 0) {
267 uv.x() = 1 - uv.x(); // mirror across u = 0.5
268 uv.y() = -uv.y(); // mirror across v = 0;
269 } else if (uv.y() > 1) {
270 uv.x() = 1 - uv.x(); // mirror across u = 0.5
271 uv.y() = 2 - uv.y(); // mirror across v = 1
272 }
273 return uv;
274}

283template<typename T>
284inline constexpr Vector<T, 3> equal_area_square_to_sphere(const Vector<T, 2>& p) {
285 T u = 2 * p.x() - 1, v = 2 * p.y() - 1;
286 T up = std::abs(u), vp = std::abs(v);
287 T signed_distance = 1 - (up + vp);
288 T d = std::abs(signed_distance);
289 T r = 1 - d;
290
291 T phi = (r == 0 ? 1 : (vp - up) / (r + 1)) * (pi_v<T> / 4);
292 T z = std::copysign(1 - r * r, signed_distance);
293
294 T cos_phi = std::copysign(std::cos(phi), u);
295 T sin_phi = std::copysign(std::sin(phi), v);
296
297 return Vector<T, 3>(
298 cos_phi * r * std::sqrt(2 - r * r),
299 sin_phi * r * std::sqrt(2 - r * r),
300 z
301 );
302}

309template<typename T>
310inline constexpr T spherical_triangle_area(
311 const Vector<T, 3>& a,
312 const Vector<T, 3>& b,
313 const Vector<T, 3>& c
314) {
315 return std::abs(T(2) * std::atan2(
316 a.dot(cross(b, c)),
317 T(1) + a.dot(b) + b.dot(c) + c.dot(a)
318 ));
