122 FilteredCameraSample sample_film_position(
123 const math::Point<int, 2>& pixel,
124 const math::Point<T, 2>& uv
125 ) const {
126 math::Point<T, 2> pixel_center(
127 static_cast<T>(pixel.x()) + T(0.5),
128 static_cast<T>(pixel.y()) + T(0.5)
129 );
130 // Sample offset according to q, then compute MC weight f/q.
131 auto offset = as_derived().sample_offset_impl(uv);
132 T f = as_derived().filter_kernel_impl(offset);
133 T q = as_derived().sample_offset_pdf_impl(offset);
134 T weight = (q > T(0)) ? (f / q) : T(0);
135 return {pixel_center + offset, offset, weight};
136 }

148 std::vector<math::Point<T, 2>> get_uv_samples(int u_sample_count, int v_sample_count) const {
149 std::vector<math::Point<T, 2>> uvs;
150 double u_stride = 1.0 / static_cast<T>(u_sample_count);
151 double v_stride = 1.0 / static_cast<T>(v_sample_count);
152 uvs.reserve(u_sample_count * v_sample_count);
153 for (int v = 0; v < v_sample_count; ++v) {
154 for (int u = 0; u < u_sample_count; ++u) {
155 T u_coord = 0.5 * u_stride + static_cast<T>(u) * u_stride;
156 T v_coord = 0.5 * v_stride + static_cast<T>(v) * v_stride;
157 uvs.emplace_back(u_coord, v_coord);
158 }
159 }
160 return uvs;
161 }

343 T filter_kernel_impl(const math::Vector<T, 2>& offset) const {
344 T ax = std::abs(offset.x()) / m_filter_radius;
345 T ay = std::abs(offset.y()) / m_filter_radius;
346 if (ax > T(1) || ay > T(1)) {
347 return T(0);
348 }
349 return (T(1) - ax) * (T(1) - ay);
