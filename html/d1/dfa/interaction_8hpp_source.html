29template<typename T>
30inline math::Point<T, 3> offset_ray_origin(
31 const math::Point<T, 3>& p_lower,
32 const math::Point<T, 3>& p_upper,
33 const Vector<T, 3>& dir,
34 const Normal<T, 3>& n
35) {
36 // Shift the ray start slightly along the geometric normal so that new rays
37 // do not immediately self‐intersect the triangle they originate from.
38 // The offset magnitude combines (1) how “thick” the triangle is when
39 // projected onto the normal and (2) a gamma-corrected term that compensates
40 // floating-point error proportional to the triangle midpoint coordinates.
41
42 auto d = abs(n.x()) * (p_upper.x() - p_lower.x()) +
43 abs(n.y()) * (p_upper.y() - p_lower.y()) +
44 abs(n.z()) * (p_upper.z() - p_lower.z());
45
46 auto p_mid = p_lower.mid(p_upper);
47
48 constexpr auto gamma7 = gamma<T>(7);
49
50 d += 2 * gamma7 * (
51 abs(p_mid.x() * abs(n.x())) +
52 abs(p_mid.y() * abs(n.y())) +
53 abs(p_mid.z() * abs(n.z()))
54 );
55
56 auto nv = n.to_vector();
57 Vector<T, 3> offset = d * nv;
58 if (is_less(nv.dot(dir), 0)) {
59 offset = -offset;
60 }
61
62 return p_mid + offset;
63}

295 auto p_from = this->point();

298 auto dist = dir.length();
299 dir = dir / dist;
