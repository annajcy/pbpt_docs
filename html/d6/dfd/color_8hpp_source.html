92template<typename T, template <typename> class Color, typename SpectrumType, typename ResponseSpectrumType>
93inline constexpr Color<T> project_spectrum(
94 const SpectrumType& spectrum,
95 const ResponseSpectrum<ResponseSpectrumType>& response
96) {
97 T r{}, g{}, b{};
98 for (int lambda = lambda_min<int>; lambda <= lambda_max<int>; ++lambda) {
99 r += spectrum.at(lambda) * response.r().at(lambda);
100 g += spectrum.at(lambda) * response.g().at(lambda);
101 b += spectrum.at(lambda) * response.b().at(lambda);
102 }
103 return Color<T> (r, g, b);
104}

112template<typename T, template <typename> class Color, typename IlluminantSpectrumType, typename ResponseSpectrumType>
113inline constexpr Color<T> project_illuminant(
114 const IlluminantSpectrumType& illuminant,
115 const ResponseSpectrum<ResponseSpectrumType>& response,
116 bool is_normalize_by_g = true
117) {
118 T r{}, g{}, b{};
119 T g_integral{};
120 for (int lambda = lambda_min<int>; lambda <= lambda_max<int>; ++lambda) {
121 g_integral += illuminant.at(lambda) * response.g().at(lambda);
122 r += illuminant.at(lambda) * response.r().at(lambda);
123 g += illuminant.at(lambda) * response.g().at(lambda);
124 b += illuminant.at(lambda) * response.b().at(lambda);
125 }
126
127 if (!is_normalize_by_g || math::is_zero(g_integral)) {
128 return Color<T> (r, g, b);

132 r / g_integral,
133 g / g_integral,
134 b / g_integral

570class LAB : public math::Vector<T, 3> {
571public:
573 LAB() : math::Vector<T, 3>(0, 0, 0) {}

577 LAB(const math::Vector<T, 3>& vec) : math::Vector<T, 3>(vec) {}
578
592 static LAB<T> from_xyz(const XYZ<T>& xyz, const XYZ<T>& white_point) {
593 auto f = [](T t) {
594 const T delta = T(6) / T(29);
595 if (t > delta * delta * delta) {
596 return std::cbrt(t);
597 } else {
598 return t / (3 * delta * delta) + T(4) / T(29);
599 }
600 };
601
602 T fx = f(xyz.x() / white_point.x());
603 T fy = f(xyz.y() / white_point.y());
604 T fz = f(xyz.z() / white_point.z());
605
606 T L = T(116) * fy - T(16);
607 T a = T(500) * (fx - fy);
608 T b = T(200) * (fy - fz);
609
610 return LAB<T>(L, a, b);
611 }
612
614 T L() const { return this->operator[](0); }
616 T a() const { return this->operator[](1); }
618 T b() const { return this->operator[](2); }
619};

765template<typename T>
766inline constexpr math::Matrix<T, 3, 3> white_balance(
767 math::Point<T, 2> src_chroma_xy,
768 math::Point<T, 2> dst_chroma_xy
769) {
770 auto src_XYZ = radiometry::XYZ<T>::from_xyY(src_chroma_xy, T{1}), dst_XYZ = radiometry::XYZ<T>::from_xyY(dst_chroma_xy, T{1});
771 auto src_LMS = radiometry::LMS<T>::from_xyz(src_XYZ), dst_LMS = radiometry::LMS<T>::from_xyz(dst_XYZ);
772 T scale_X = dst_LMS.x() / src_LMS.x();
773 T scale_Y = dst_LMS.y() / src_LMS.y();
774 T scale_Z = dst_LMS.z() / src_LMS.z();
775 math::Matrix<T, 3, 3> wb_matrix = math::Matrix<T, 3, 3>::zeros();
776 wb_matrix[0][0] = scale_X;
777 wb_matrix[1][1] = scale_Y;
778 wb_matrix[2][2] = scale_Z;
779 return radiometry::LMS<T>::lms_to_xyz_matrix() *
780 wb_matrix *
781 radiometry::LMS<T>::xyz_to_lms_matrix();
782}
