139 struct Pixel {
141 std::array<T, 3> rgb_sum{0.0, 0.0, 0.0};
143 double weight_sum{0.0};
144
151 void add_sample(const radiometry::RGB<T>& rgb, T weight) {
152 double w = static_cast<double>(weight);
153 rgb_sum[0] += w * static_cast<double>(rgb[0]);
154 rgb_sum[1] += w * static_cast<double>(rgb[1]);
155 rgb_sum[2] += w * static_cast<double>(rgb[2]);
156 weight_sum += w;
157 }
158
168 radiometry::RGB<T> resolve() const {
169 if (math::is_zero(weight_sum)) {
170 return radiometry::RGB<T>(T(0), T(0), T(0));
171 }
172 double inv_weight = 1.0 / weight_sum;
173 return radiometry::RGB<T>(
174 static_cast<T>(rgb_sum[0] * inv_weight),
175 static_cast<T>(rgb_sum[1] * inv_weight),
176 static_cast<T>(rgb_sum[2] * inv_weight)
177 );
178 }
179
181 void clear() {
182 rgb_sum = {0.0, 0.0, 0.0};
183 weight_sum = 0.0;
184 }
185 };
