70template<typename T, typename LuminantSpectrumType>
71inline RGBSigmoidPolynomialOptimizationResult<T> optimize_albedo_rgb_sigmoid_polynomial(
72 const RGB<T>& target_rgb,
73 const RGBColorSpace<T>& color_space,
74 const LuminantSpectrumType& reference_luminant_spectrum,
75 int max_iterations = 300,
76 double learning_rate = 1.0,
77 double delta_x = 1e-4,
78 double error_threshold = 1e-2,
79 const std::array<T, 3>& initial_guess = {0.0, 0.0, 0.0},
80 bool verbose = false
81) {
82
83 if (verbose) {
84 std::cout << "Starting optimization for target RGB: " << target_rgb << std::endl;
85 }
86 XYZ<T> target_xyz = color_space.to_xyz(target_rgb);
87 if (verbose) {
88 std::cout << "Target XYZ: " << target_xyz << std::endl;
89 }
90 LAB<T> target_lab = LAB<T>::from_xyz(target_xyz, color_space.white_point());
91 if (verbose) {
92 std::cout << "Target LAB: " << target_lab << std::endl;
93 }
94
95 auto eval_lab = [&](const std::array<T,3>& c) -> LAB<T> {
96 RGBAlbedoSpectrumDistribution<T, RGBSigmoidPolynomialNormalized> albedo({c[0], c[1], c[2]});
97 XYZ<T> xyz = XYZ<T>::from_reflectance(albedo, reference_luminant_spectrum);
98 return LAB<T>::from_xyz(xyz, color_space.white_point());
99 };
100
101 auto eval_error = [&](const LAB<T>& lab) -> math::Vector<T, 3> {
102 T dL = target_lab.L() - lab.L();
103 T da = target_lab.a() - lab.a();
104 T db = target_lab.b() - lab.b();
105 return math::Vector<T, 3>{dL, da, db};
106 };
107
108 auto eval_jacobian = [&](const std::array<T,3>& c) -> math::Matrix<T, 3, 3> {
109 const T epsilon = delta_x;
110 math::Matrix<T, 3, 3> J;
111 for (int i = 0; i < 3; i++) {
112 std::array<T, 3> c_plus = c;
113 std::array<T, 3> c_minus = c;
114 c_plus[i] += epsilon;
115 c_minus[i] -= epsilon;
116 auto err_plus = eval_error(eval_lab(c_plus));
117 auto err_minus = eval_error(eval_lab(c_minus));
118 J[0][i] = (err_plus[0] - err_minus[0]) / (2 * epsilon);
119 J[1][i] = (err_plus[1] - err_minus[1]) / (2 * epsilon);
120 J[2][i] = (err_plus[2] - err_minus[2]) / (2 * epsilon);
121 }
122 return J;
123 };
124
125 bool converged = false;
126 std::array<T, 3> coeffs = initial_guess;
127 auto error = math::Vector<T, 3>::filled(std::numeric_limits<double>::max());
128 for (int i = 0; i < max_iterations; i ++) {
129 auto object_lab = eval_lab(coeffs);
130 //std::cout << "Object LAB: " << object_lab << std::endl;
131 error = eval_error(object_lab);
132
133 if (verbose) {
134 std::cout << "Iteration " << i << ": Error = " << error
135 << ", Coeffs = (" << coeffs[0] << ", " << coeffs[1] << ", " << coeffs[2] << ")\n";
136 }
137
138 if ([&]() -> bool {
139 for (int j = 0; j < 3; j++) {
140 if (!math::is_less_equal(math::abs(error[j]), error_threshold)) {
141 return false;
142 }
143 }
144 return true;
145 }()) {
146 if (verbose) {
147 std::cout << "Converged after " << i << " iterations.\n";
148 std::cout << "Final Coeffs = (" << coeffs[0] << ", " << coeffs[1] << ", " << coeffs[2] << ")\n";
149 } else {
150 std::cout << "converged after " << i << " iterations.\n";
151 }
152 converged = true;
153 return RGBSigmoidPolynomialOptimizationResult<T>{error, coeffs};
154 }
155
156 auto J = eval_jacobian(coeffs);
157 if (verbose) {
158 std::cout << "Jacobian:\n" << J << std::endl;
159 }
160
161 // Solve the linear system J * delta = error and update coeffs
162 auto delta = J.inversed_rref() * error;
163 coeffs[0] -= delta[0] * learning_rate;
164 coeffs[1] -= delta[1] * learning_rate;
165 coeffs[2] -= delta[2] * learning_rate;
166 }
167
168 if (!converged) {
169 std::cout << "Did not converge after " << max_iterations << " iterations.\n";
170 std::cout << "Final Error = " << error << "\n";
171 std::cout << "Final Coeffs = (" << coeffs[0] << ", " << coeffs[1] << ", " << coeffs[2] << ")\n";
172 }
173
174 return RGBSigmoidPolynomialOptimizationResult<T>{error, coeffs};
175}
