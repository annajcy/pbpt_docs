178 Vector<T, 3> f = (target - eye).normalized();

180 Vector<T, 3> u = cross(s, f);
181 Matrix<T, 4, 4> m(

183 u.x(), u.y(), u.z(), -u.dot(eye.to_vector()),

185 T(0), T(0), T(0), T(1)

443template<typename T>
444TransformDecompositionResult<T> fast_decompose_transform(const Transform<T>& transform) {
445 const auto& M4 = transform.matrix(); // 4x4
446 using Mat3 = Matrix<T,3,3>;
447 using Vec3 = Vector<T,3>;
448
449 // --- 1) Translation ---
450 const Vec3 t(M4.at(0,3), M4.at(1,3), M4.at(2,3));
451 Transform<T> Txf = Transform<T>::translate(t);
452
453 // --- 2) A = upper-left 3x3 ---
454 MatrixView<T, 4, 4, 3, 3> Aview{ M4, 0, 0 };
455 Matrix<T, 3, 3> A = Aview.to_matrix(); // A = R * S (no shear assumed)
456
457 // --- 3) scales = column norms ---
458 Vector<T, 3> s{};
459 const T eps = T(1e-12);
460 for (int c = 0; c < 3; ++c) {
461 // VectorView /=
462 T n2 = T(0);
463 for (int r = 0; r < 3; ++r) n2 += A.at(r,c) * A.at(r,c);
464 T n = std::sqrt(n2);
465 s[c] = (n > eps) ? n : T(1); //
466 }
467
468 // --- 4) R = A * S^{-1} ---
469 Matrix<T, 3, 3> R = A;
470 for (int c = 0; c < 3; ++c) {
471 T invs = T(1) / s[c];
472 for (int r = 0; r < 3; ++r) R.at(r,c) *= invs;
473 }
474
475 // --- 5) GramSchmidt ---
476 auto normalize_col = [&](int c){
477 T n2 = T(0); for (int r=0;r<3;++r) n2 += R.at(r,c)*R.at(r,c);
478 T n = std::sqrt(n2); if (n > eps) for (int r=0;r<3;++r) R.at(r,c) /= n;
479 };
480 // GramSchmidt: c0, c1, c2
481 // c1 := c1 - proj_{c0}(c1)
482 auto dot_col = [&](int i, int j){
483 T s = T(0); for (int r=0;r<3;++r) s += R.at(r,i)*R.at(r,j); return s;
484 };
485 auto axpy_col = [&](int dst, int src, T alpha){
486 for (int r=0;r<3;++r) R.at(r,dst) -= alpha * R.at(r,src);
487 };
488 normalize_col(0);
489 T a10 = dot_col(1,0); axpy_col(1,0,a10); normalize_col(1);
490 // c2 := c2 - proj_{c0}(c2) - proj_{c1}(c2)
491 T a20 = dot_col(2,0); axpy_col(2,0,a20);
492 T a21 = dot_col(2,1); axpy_col(2,1,a21);
493 normalize_col(2);
494
495 // --- 6) det(R) < 0 ---
496 T detR = R.determinant();
497 if (detR < T(0)) {
498 // det(R) -> +1
499 int k = 0;
500 if (std::abs(s[1]) > std::abs(s[k])) k = 1;
501 if (std::abs(s[2]) > std::abs(s[k])) k = 2;
502 s[k] = -s[k];
503 for (int r = 0; r < 3; ++r) R.at(r,k) = -R.at(r,k);
504 }
505
506 // --- 7) ---
507 Transform<T> Rxf = Transform<T>::from_mat3x3(R);
508 Transform<T> Sxf = Transform<T>::scale(s);
509 return { Txf, Rxf, Sxf };
510}
