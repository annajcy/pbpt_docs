125static constexpr Mat4 rotate(Float angle_rad, const Vec3& axis) noexcept {
126 // 确保旋转轴是单位向量，这对于公式的正确性至关重要
127 const Vec3 a = axis.normalized();
128 const Float s = sin(angle_rad);
129 const Float c = cos(angle_rad);
130 const Float omc = 1.0f - c; // one-minus-cosine
131
132 const Float ax = a.x();
133 const Float ay = a.y();
134 const Float az = a.z();
135
136 // 罗德里格斯旋转公式的矩阵形式
137 return Mat4(
138 c + ax * ax * omc, ax * ay * omc - az * s, ax * az * omc + ay * s, 0,
139 ay * ax * omc + az * s, c + ay * ay * omc, ay * az * omc - ax * s, 0,
140 az * ax * omc - ay * s, az * ay * omc + ax * s, c + az * az * omc, 0,
141 0, 0, 0, 1
142 );
143}

155 static constexpr Mat4 look_at(const Pt3& eye, const Pt3& target, const Vec3& up) noexcept {
156 const Vec3 f = (target - eye).normalized(); // Forward vector
157 const Vec3 s = f.cross(up).normalized(); // Right vector
158 const Vec3 u = s.cross(f); // Recalculated Up vector
159
160 return Mat4(
161 s.x(), s.y(), s.z(), -s.dot(eye.to_vector()),
162 u.x(), u.y(), u.z(), -u.dot(eye.to_vector()),
163 -f.x(), -f.y(), -f.z(), f.dot(eye.to_vector()),
164 0, 0, 0, 1
165 );
166 }
