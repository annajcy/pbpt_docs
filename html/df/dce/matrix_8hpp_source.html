477 constexpr T determinant() const
478 requires(R == C)
479 {
480 if constexpr (R == 1) {
481 return (*this).at(0, 0);
482 } else if constexpr (R == 2) {
483 return (*this).at(0, 0) * (*this).at(1, 1) - (*this).at(0, 1) * (*this).at(1, 0);
484 } else if constexpr (R == 3) {
485 return (*this).at(0, 0) * ((*this).at(1, 1) * (*this).at(2, 2) - (*this).at(1, 2) * (*this).at(2, 1)) -
486 (*this).at(0, 1) * ((*this).at(1, 0) * (*this).at(2, 2) - (*this).at(1, 2) * (*this).at(2, 0)) +
487 (*this).at(0, 2) * ((*this).at(1, 0) * (*this).at(2, 1) - (*this).at(1, 1) * (*this).at(2, 0));
488 } else if constexpr (R == 4) {
489 // Optimized 4x4 determinant using cofactor expansion along first row
490 return (*this).at(0, 0) * minor_matrix(0, 0).determinant() -
491 (*this).at(0, 1) * minor_matrix(0, 1).determinant() +
492 (*this).at(0, 2) * minor_matrix(0, 2).determinant() -
493 (*this).at(0, 3) * minor_matrix(0, 3).determinant();
494 } else {
495 // General case: Laplace expansion (less efficient, but works for
496 // any size)
497 T det = 0;
498 for (int c = 0; c < C; ++c) {
499 T sign = (c % 2 == 0) ? 1 : -1;
500 det += sign * (*this).at(0, c) * minor_matrix(0, c).determinant();
501 }
502 return det;
503 }

515 requires(R == C)

517 T det = determinant();
518 assert_if_ex<std::domain_error>([&det]() { return is_equal(det, T(0)); },

523 Matrix result{};

614 constexpr Matrix inverse_2x2_optimized() const
615 requires(R == 2 && C == 2)
616 {
617 T det = (*this).determinant();
618 T inv_det = T(1) / det;
619 Matrix result{};
620
621 // For 2x2 matrix [[a, b], [c, d]], inverse is (1/det) * [[d, -b], [-c, a]]
622 result.at(0, 0) = (*this).at(1, 1) * inv_det; // d
623 result.at(0, 1) = -(*this).at(0, 1) * inv_det; // -b
624 result.at(1, 0) = -(*this).at(1, 0) * inv_det; // -c
625 result.at(1, 1) = (*this).at(0, 0) * inv_det; // a
626
627 return result;
628 }

661 constexpr Matrix inverse_4x4_optimized() const
662 requires(R == 4 && C == 4)
663 {
664 // Use direct analytical formula for 4x4 matrix inversion
665 // More efficient than cofactor expansion
666 const T a00 = (*this).at(0, 0), a01 = (*this).at(0, 1), a02 = (*this).at(0, 2), a03 = (*this).at(0, 3);
667 const T a10 = (*this).at(1, 0), a11 = (*this).at(1, 1), a12 = (*this).at(1, 2), a13 = (*this).at(1, 3);
668 const T a20 = (*this).at(2, 0), a21 = (*this).at(2, 1), a22 = (*this).at(2, 2), a23 = (*this).at(2, 3);
669 const T a30 = (*this).at(3, 0), a31 = (*this).at(3, 1), a32 = (*this).at(3, 2), a33 = (*this).at(3, 3);
670
671 // Calculate 2x2 determinants for cofactor computation
672 const T det2_01_01 = a00 * a11 - a01 * a10;
673 const T det2_01_02 = a00 * a12 - a02 * a10;
674 const T det2_01_03 = a00 * a13 - a03 * a10;
675 const T det2_01_12 = a01 * a12 - a02 * a11;
676 const T det2_01_13 = a01 * a13 - a03 * a11;
677 const T det2_01_23 = a02 * a13 - a03 * a12;
678
679 const T det2_23_01 = a20 * a31 - a21 * a30;
680 const T det2_23_02 = a20 * a32 - a22 * a30;
681 const T det2_23_03 = a20 * a33 - a23 * a30;
682 const T det2_23_12 = a21 * a32 - a22 * a31;
683 const T det2_23_13 = a21 * a33 - a23 * a31;
684 const T det2_23_23 = a22 * a33 - a23 * a32;
685
686 // Calculate 3x3 cofactors for adjugate matrix
687 const T cof00 = +(a11 * det2_23_23 - a12 * det2_23_13 + a13 * det2_23_12);
688 const T cof01 = -(a10 * det2_23_23 - a12 * det2_23_03 + a13 * det2_23_02);
689 const T cof02 = +(a10 * det2_23_13 - a11 * det2_23_03 + a13 * det2_23_01);
690 const T cof03 = -(a10 * det2_23_12 - a11 * det2_23_02 + a12 * det2_23_01);
691
692 const T cof10 = -(a01 * det2_23_23 - a02 * det2_23_13 + a03 * det2_23_12);
693 const T cof11 = +(a00 * det2_23_23 - a02 * det2_23_03 + a03 * det2_23_02);
694 const T cof12 = -(a00 * det2_23_13 - a01 * det2_23_03 + a03 * det2_23_01);
695 const T cof13 = +(a00 * det2_23_12 - a01 * det2_23_02 + a02 * det2_23_01);
696
697 const T cof20 = +(a31 * det2_01_23 - a32 * det2_01_13 + a33 * det2_01_12);
698 const T cof21 = -(a30 * det2_01_23 - a32 * det2_01_03 + a33 * det2_01_02);
699 const T cof22 = +(a30 * det2_01_13 - a31 * det2_01_03 + a33 * det2_01_01);
700 const T cof23 = -(a30 * det2_01_12 - a31 * det2_01_02 + a32 * det2_01_01);
701
702 const T cof30 = -(a21 * det2_01_23 - a22 * det2_01_13 + a23 * det2_01_12);
703 const T cof31 = +(a20 * det2_01_23 - a22 * det2_01_03 + a23 * det2_01_02);
704 const T cof32 = -(a20 * det2_01_13 - a21 * det2_01_03 + a23 * det2_01_01);
705 const T cof33 = +(a20 * det2_01_12 - a21 * det2_01_02 + a22 * det2_01_01);
706
707 // Calculate determinant using first row
708 const T det = a00 * cof00 + a01 * cof01 + a02 * cof02 + a03 * cof03;
709
710 assert_if_ex<std::domain_error>([&det]() { return is_equal(det, T(0)); },
711 "Cannot invert a singular 4x4 matrix");
712
713 const T inv_det = T(1) / det;
714
715 // Build result matrix (transposed cofactor matrix / determinant)
716 Matrix result{};
717 result.at(0, 0) = cof00 * inv_det; result.at(0, 1) = cof10 * inv_det;
718 result.at(0, 2) = cof20 * inv_det; result.at(0, 3) = cof30 * inv_det;
719 result.at(1, 0) = cof01 * inv_det; result.at(1, 1) = cof11 * inv_det;
720 result.at(1, 2) = cof21 * inv_det; result.at(1, 3) = cof31 * inv_det;
721 result.at(2, 0) = cof02 * inv_det; result.at(2, 1) = cof12 * inv_det;
722 result.at(2, 2) = cof22 * inv_det; result.at(2, 3) = cof32 * inv_det;

1031 //transform to row echelon form and reduced row echelon form, return rank and number of row swaps
1039 constexpr EchelonResult ref_inplace() {
1040 int r = 0;
1041 int swaps = 0;
1042 T total_scale_inv = T(1);
1043 for (int c = 0; c < C && r < R; c ++) {
1044 auto [pr, pv] = argmax_abs_in_col(c, r);
1045 if (is_zero(pv)) { continue; }
1046 if (pr != r) swap_rows(r, pr), swaps ++;
1047 scale_row(r, T(1) / pv), total_scale_inv *= pv;
1048 for (int rr = r + 1; rr < R; rr ++) {
1049 T s = -(*this).at(rr, c);
1050 add_scaled_row(rr, r, s);
1051 }
1052 r ++;
1053 }
1054 visit([&](T& x, int, int){ if (is_zero(x)) x = T(0); });
1055 return {(*this), r, swaps, total_scale_inv};
1056 }

1174// slove least mean square problem: find M that minimizes ||M * A - B||
1175template <typename T, int N, int ConstraintsCount>
1176inline constexpr Matrix<T, N, N> solve_LMS(
1177 const Matrix<T, N, ConstraintsCount>& A,
1178 const Matrix<T, N, ConstraintsCount>& B
1179) {
1180 Matrix<T, N, N> result = Matrix<T, N, N>::zeros();
1181 auto At = A.transposed();
1182 auto AAt = A * At;
1183 auto BAt = B * At;
1184 auto AAt_inv = AAt.inversed_rref();
1185 result = BAt * AAt_inv;
1186 return result;
1187}
