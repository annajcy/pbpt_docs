518 constexpr T determinant() const requires(R == C) {
519 if constexpr (R == 1) {
520 return (*this).at(0, 0);
521 } else if constexpr (R == 2) {
522 return (*this).at(0, 0) * (*this).at(1, 1) - (*this).at(0, 1) * (*this).at(1, 0);
523 } else if constexpr (R == 3) {
524 return (*this).at(0, 0) * ((*this).at(1, 1) * (*this).at(2, 2) - (*this).at(1, 2) * (*this).at(2, 1)) -
525 (*this).at(0, 1) * ((*this).at(1, 0) * (*this).at(2, 2) - (*this).at(1, 2) * (*this).at(2, 0)) +
526 (*this).at(0, 2) * ((*this).at(1, 0) * (*this).at(2, 1) - (*this).at(1, 1) * (*this).at(2, 0));
527 } else {
528 // General case: Laplace expansion (less efficient, but works for any size)
529 T det = 0;
530 for (int c = 0; c < C; ++c) {
531 T sign = (c % 2 == 0) ? 1 : -1;
532 det += sign * (*this).at(0, c) * submatrix(0, c).determinant();
533 }
534 return det;
535 }

543 constexpr Matrix inversed() const requires(R == C) {
544 T det = determinant();
545 if (det == 0) {
546 if (std::is_constant_evaluated()) {
547 throw "Compile-time error: Cannot invert a singular matrix";
548 } else {
549 throw std::runtime_error("Cannot invert a singular matrix");

553 Matrix cofactor_matrix{};
554 for (int r = 0; r < R; ++r) {
555 for (int c = 0; c < C; ++c) {
556 T sign = ((r + c) % 2 == 0) ? 1 : -1;
557 cofactor_matrix[r][c] = sign * submatrix(r, c).determinant();
558 }
559 }
560
561 Matrix adjugate_matrix = cofactor_matrix.transpose();
562 return adjugate_matrix * (static_cast<T>(1.0) / det);

607 constexpr auto submatrix(int row_to_remove, int col_to_remove) const requires(R == C) {
608 Matrix<T, R - 1, C - 1> sub{};
609 int sub_r = 0;
610 for (int r = 0; r < R; ++r) {
611 if (r == row_to_remove) continue;
612 int sub_c = 0;
613 for (int c = 0; c < C; ++c) {
614 if (c == col_to_remove) continue;
615 sub[sub_r][sub_c] = (*this)[r][c];
616 sub_c++;
617 }
618 sub_r++;
