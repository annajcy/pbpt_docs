517 constexpr T determinant() const requires(R == C) {
518 if constexpr (R == 1) {
519 return (*this)(0, 0);
520 } else if constexpr (R == 2) {
521 return (*this)(0, 0) * (*this)(1, 1) - (*this)(0, 1) * (*this)(1, 0);
522 } else if constexpr (R == 3) {
523 return (*this)(0, 0) * ((*this)(1, 1) * (*this)(2, 2) - (*this)(1, 2) * (*this)(2, 1)) -
524 (*this)(0, 1) * ((*this)(1, 0) * (*this)(2, 2) - (*this)(1, 2) * (*this)(2, 0)) +
525 (*this)(0, 2) * ((*this)(1, 0) * (*this)(2, 1) - (*this)(1, 1) * (*this)(2, 0));
526 } else {
527 // General case: Laplace expansion (less efficient, but works for any size)
528 T det = 0;
529 for (int c = 0; c < C; ++c) {
530 T sign = (c % 2 == 0) ? 1 : -1;
531 det += sign * (*this)(0, c) * submatrix(0, c).determinant();
532 }
533 return det;
534 }
535 }

542 constexpr Matrix inverse() const requires(R == C) {
543 T det = determinant();
544 if (det == 0) {
545 if (std::is_constant_evaluated()) {
546 throw "Compile-time error: Cannot invert a singular matrix";
547 } else {
548 throw std::runtime_error("Cannot invert a singular matrix");
549 }
550 }
551
552 Matrix cofactor_matrix{};
553 for (int r = 0; r < R; ++r) {
554 for (int c = 0; c < C; ++c) {
555 T sign = ((r + c) % 2 == 0) ? 1 : -1;
556 cofactor_matrix(r, c) = sign * submatrix(r, c).determinant();
557 }
558 }
559
560 Matrix adjugate_matrix = cofactor_matrix.transpose();
561 return adjugate_matrix * (static_cast<T>(1.0) / det);
