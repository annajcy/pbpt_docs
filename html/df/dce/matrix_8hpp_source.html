507 constexpr T determinant() const requires(R == C) {
508 if constexpr (R == 1) {
509 return (*this)(0, 0);
510 } else if constexpr (R == 2) {
511 return (*this)(0, 0) * (*this)(1, 1) - (*this)(0, 1) * (*this)(1, 0);
512 } else if constexpr (R == 3) {
513 return (*this)(0, 0) * ((*this)(1, 1) * (*this)(2, 2) - (*this)(1, 2) * (*this)(2, 1)) -
514 (*this)(0, 1) * ((*this)(1, 0) * (*this)(2, 2) - (*this)(1, 2) * (*this)(2, 0)) +
515 (*this)(0, 2) * ((*this)(1, 0) * (*this)(2, 1) - (*this)(1, 1) * (*this)(2, 0));
516 } else {
517 // General case: Laplace expansion (less efficient, but works for any size)
518 T det = 0;
519 for (int c = 0; c < C; ++c) {
520 T sign = (c % 2 == 0) ? 1 : -1;
521 det += sign * (*this)(0, c) * submatrix(0, c).determinant();
522 }
523 return det;
524 }
525 }

532 constexpr Matrix inverse() const requires(R == C) {
533 T det = determinant();
534 if (det == 0) {
535 if (std::is_constant_evaluated()) {
536 throw "Compile-time error: Cannot invert a singular matrix";
537 } else {
538 throw std::runtime_error("Cannot invert a singular matrix");

542 Matrix cofactor_matrix{};
543 for (int r = 0; r < R; ++r) {
544 for (int c = 0; c < C; ++c) {
545 T sign = ((r + c) % 2 == 0) ? 1 : -1;
546 cofactor_matrix(r, c) = sign * submatrix(r, c).determinant();
547 }
548 }

550 Matrix adjugate_matrix = cofactor_matrix.transpose();
551 return adjugate_matrix * (static_cast<T>(1.0) / det);
