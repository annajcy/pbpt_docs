122 SampledSpectrumType trace_ray(
123 const geometry::Ray<T, 3>& ray,
124 const SampledWavelengthType& wavelengths
125 ) const {
126 T closest = std::numeric_limits<T>::infinity();
127 std::optional<geometry::SurfaceInteraction<T>> closest_hit{};
128
129 int shape_index = -1;
130 for (const auto& obj : m_scene_objects) {
131 if (auto hit = obj.sphere.intersect(ray)) {
132 const auto& [si, t_hit] = *hit;
133 if (t_hit < closest) {
134 closest = t_hit;
135 closest_hit = si;
136 shape_index = &obj - &m_scene_objects[0];
137 }
138 }
139 }
140
141 if (closest_hit.has_value()) {
142 return shade_hit(closest_hit.value(), ray, wavelengths, shape_index);
143 }
144
145 return shade_background(ray.direction(), wavelengths);
146 }
147
148 SampledSpectrumType shade_hit(
149 const geometry::SurfaceInteraction<T>& si,
150 const geometry::Ray<T, 3>& ray,
151 const SampledWavelengthType& wavelengths,
152 const int sphere_index
153 ) const {
154 auto normal = si.n().to_vector().normalized();
155 const T intensity = std::max(T(0), normal.dot(-ray.direction()));
156 auto albedo_sample = m_sphere_albedo_spectra[sphere_index].sample(wavelengths);
157 auto illuminant_sample = m_scene_illuminant.sample(wavelengths);
158 return albedo_sample * illuminant_sample * intensity;
159 }
160
161 SampledSpectrumType shade_background(
162 const math::Vector<T, 3>& dir,
163 const SampledWavelengthType& wavelengths
164 ) const {
165 auto background = radiometry::constant::get_swatch_reflectance<T>(m_background_reflectance);
166 return background.sample(wavelengths) * m_scene_illuminant.sample(wavelengths);
167 }
