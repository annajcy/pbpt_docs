104 rgb_camera[0][i] = rgb[0];
105 rgb_camera[1][i] = rgb[1];
106 rgb_camera[2][i] = rgb[2];

109 math::Matrix<T, 3, radiometry::constant::swatch_reflectances_count> xyz_output{};
110 for (int i = 0; i < radiometry::constant::swatch_reflectances_count; ++i) {
111 auto reflectance = radiometry::constant::swatch_reflectances<T>[i];
112 auto xyz = radiometry::project_reflectance<T, radiometry::XYZ>(
113 reflectance,
114 m_standard_illuminant,

116 radiometry::constant::CIE_X<T>,
117 radiometry::constant::CIE_Y<T>,
118 radiometry::constant::CIE_Z<T>
119 ),
120 false);
121 xyz_output[0][i] = xyz[0];
122 xyz_output[1][i] = xyz[1];
123 xyz_output[2][i] = xyz[2];
124 }

126 m_sensor_rgb_to_xyz = math::solve_LMS(
127 rgb_camera,
128 xyz_output
