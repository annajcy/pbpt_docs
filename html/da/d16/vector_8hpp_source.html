69template <std::floating_point T>
70constexpr std::pair<Vector<T, 3>, Vector<T, 3>> coordinate_system(const Vector<T, 3>& v1) {
71 assert_if([&v1]() { return !v1.is_normalized(); }, "Input vector to coordinate_system_stable() must be normalized");
72
73 const T sign = std::copysign(T(1), v1.z());
74 const T a = T(-1) / (sign + v1.z());
75 const T b = v1.x() * v1.y() * a;
76
77 Vector<T, 3> v2(T(1) + sign * v1.x() * v1.x() * a, sign * b, -sign * v1.x());
78
79 Vector<T, 3> v3(b, sign + v1.y() * v1.y() * a, -v1.y());
80
81 return {v2.normalized(), v3.normalized()};
82}
