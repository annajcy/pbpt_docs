54template <typename T>
55constexpr promote_int_to_float_t<T> angle_between(const Vector<T, 3>& v1, const Vector<T, 3>& v2) {
56 assert_if([&v1, &v2]() { return v1.is_all_zero() || v2.is_all_zero(); }, "Cannot compute angle between zero vectors");
57 assert_if([&v1, &v2]() { return !v1.is_normalized() || !v2.is_normalized(); },
58 "Vectors must be normalized to compute angle between them");
59 if (v1.dot(v2) < 0) {
60 return pi_v<T> - 2 * safe_asin((v1 + v2).length() / 2);
61 } else {
62 return 2 * safe_asin((v1 - v2).length() / 2);
63 }
64}

73template <std::floating_point T>
74constexpr std::pair<Vector<T, 3>, Vector<T, 3>> coordinate_system(const Vector<T, 3>& v1) {
75 assert_if([&v1]() { return !v1.is_normalized(); }, "Input vector to coordinate_system_stable() must be normalized");
76
77 const T sign = std::copysign(T(1), v1.z());
78 const T a = T(-1) / (sign + v1.z());
79 const T b = v1.x() * v1.y() * a;
80
81 Vector<T, 3> v2(T(1) + sign * v1.x() * v1.x() * a, sign * b, -sign * v1.x());
82
83 Vector<T, 3> v3(b, sign + v1.y() * v1.y() * a, -v1.y());
84
85 return {v2.normalized(), v3.normalized()};
86}
