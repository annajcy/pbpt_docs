198template<typename T>
199class BlackBodySpectrumDistribution : public SpectrumDistribution<BlackBodySpectrumDistribution<T>, T> {
200 friend class SpectrumDistribution<BlackBodySpectrumDistribution<T>, T>;
201public:
203 constexpr BlackBodySpectrumDistribution(T temperature) : m_temperature(temperature) {}
213 constexpr static inline T black_body(T t_K, T lambda_nm) {
214 // Planck's law formula for black body radiation
215 const double h = 6.62607015e-34; // Planck's constant
216 const double c = 299792458; // Speed of light
217 const double k = 1.380649e-23; // Boltzmann's constant
218 double l = lambda_nm * 1e-9; // Convert nm to m
219 double L = (2 * h * c * c) / math::pow(l, 5) * (1 / (math::fast_exp(h * c / (l * k * t_K)) - 1));
220 return L;
221 }
222
230 constexpr static inline T black_body_M(T t_K) {
231 // Stefan-Boltzmann law
232 const double sigma = 5.670374419e-8; // Stefan-Boltzmann constant
233 double E = sigma * std::pow(t_K, 4);
234 return E;
235 }
236
237
244 constexpr static inline T non_black_body(T t_K, T lambda_nm, T p_hd) {
245 // Kirchhoff's law of thermal radiation
246 return black_body(t_K, lambda_nm) * (1 - p_hd);
247 }
248
259 constexpr static inline T black_body_max_wavelength(T t_K) {
260 // Wien's displacement law
261 const double b = 2.897771955e-3; // Wien's displacement constant
262 return b / t_K * 1e9; // Convert from m to nm
263 }
264
265private:
267 T m_temperature;
268
269public:
270 constexpr T max_wavelength() const {
271 return black_body_max_wavelength(m_temperature);
272 }
273
274 constexpr T max_value() const {
275 return black_body(m_temperature, max_wavelength());
276 }
277
278private:
279 constexpr T at_impl(T lambda) const {
280 return black_body(m_temperature, lambda);
281 }
282};

314template<typename T>
315class PiecewiseLinearSpectrumDistribution : public SpectrumDistribution<PiecewiseLinearSpectrumDistribution<T>, T> {
316 friend class SpectrumDistribution<PiecewiseLinearSpectrumDistribution<T>, T>;
317private:
318 std::vector<std::pair<T, T>> m_points; // (lambda, value)
319
320public:
326 PiecewiseLinearSpectrumDistribution(const std::vector<std::pair<T, T>>& points)
327 : m_points(points) {
328 sort_points();
329 }
330
331private:
332 constexpr T at_impl(T lambda) const {
333 if (m_points.empty()) {
334 return T(0);
335 }
336
337 if (lambda <= m_points.front().first) {
338 return m_points.front().second;
339 }
340
341 if (lambda >= m_points.back().first) {
342 return m_points.back().second;
343 }
344
345 auto it = std::lower_bound(
346 m_points.begin(),
347 m_points.end(),
348 lambda,
349 [](const auto& point, T value) { return point.first < value; }
350 );
351
352 const auto& [lambda1, value1] = *it;
353 const auto& [lambda0, value0] = *(it - 1);
354
355 return value0 + (value1 - value0) * (lambda - lambda0) / (lambda1 - lambda0);
356 }
357
358 void sort_points() {
359 std::sort(m_points.begin(), m_points.end(), [](const auto& a, const auto& b) {
360 return a.first < b.first;
361 });
362 }
363
370 void add_point(T lambda, T value) {
371 m_points.emplace_back(lambda, value);
372 }
373
379 void remove_point(T lambda) {
380 std::erase_if(m_points, [lambda](const auto& p) {
381 return p.first == lambda;
382 });
383 }
384};

508 T c0;
510 T c1;

517 RGBSigmoidPolynomialNormalized(const std::array<T, 3>& coeffs) : c0(coeffs[0]), c1(coeffs[1]), c2(coeffs[2]) {}
518
528 constexpr T at(T lambda) const {
529 const T t = (lambda - T(360)) / T(830 - 360); //
530 return math::sigmoid(math::Polynomial<T>::evaluate(t, c0, c1, c2));
531 }
532
540 RGBSigmoidPolynomial<T> to_unnormalized() const {
541 // : t = (lambda - 360) / (830 - 360)
542 // : A*t^2 + B*t + C
543 // wavelength: A'*lambda^2 + B'*lambda + C'
544 double c0_val = 360.0, c1_val = 1.0 / (830.0 - 360.0);
545 double A = c2, B = c1, C = c0;
546 double c1_squared = c1_val * c1_val;
547 double c0_c1 = c0_val * c1_val;
548
549 return RGBSigmoidPolynomial<T>{
550 T(C - B * c0_val * c1_val + A * c0_c1 * c0_c1),
551 T(B * c1_val - 2 * A * c0_val * c1_squared),
552 T(A * c1_squared)
553 };
554 }
