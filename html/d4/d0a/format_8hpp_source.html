19template <template <typename, int> typename Derived, typename T, int N>
20std::string tuple_name() {
21 if constexpr (std::is_same_v<Derived<T, N>, Vector<T, N>>) {
22 return std::format("Vector<{}, {}>", typeid(T).name(), N);
23 } else if constexpr (std::is_same_v<Derived<T, N>, Point<T, N>>) {
24 return std::format("Point<{}, {}>", typeid(T).name(), N);
25 } else if constexpr (std::is_same_v<Derived<T, N>, Normal<T, N>>) {
26 return std::format("Normal<{}, {}>", typeid(T).name(), N);
27 } else if constexpr (std::is_same_v<Derived<T, N>, Homogeneous<T, N>>) {
28 return std::format("Homogeneous<{}, {}>", typeid(T).name(), N);
29 } else {
30 return "Unknown";
31 }
32}

34template <template <typename, int> typename Derived, typename T, int N>
35std::string to_string(const Tuple<Derived, T, N>& t) {
36 std::ostringstream oss;
37 oss << std::fixed << std::setprecision(6);
38 oss << tuple_name<Derived, T, N>() << "(";
39 for (int i = 0; i < N; ++i) {
40 oss << t[i] << (i == N - 1 ? "" : ", ");
41 }
42 oss << ")";
43 return oss.str();

54 oss << std::format("Matrix<{}, {}, {}>(", typeid(T).name(), R, C);
55 oss << "\n";
56 for (int r = 0; r < R; ++r) {
