29template <template <typename, int> typename Derived, typename T, int N>
30std::string tuple_name() {
31 if constexpr (std::is_same_v<Derived<T, N>, Vector<T, N>>) {
32 return std::format("Vector<{}, {}>", typeid(T).name(), N);
33 } else if constexpr (std::is_same_v<Derived<T, N>, Point<T, N>>) {
34 return std::format("Point<{}, {}>", typeid(T).name(), N);
35 } else if constexpr (std::is_same_v<Derived<T, N>, Normal<T, N>>) {
36 return std::format("Normal<{}, {}>", typeid(T).name(), N);
37 } else if constexpr (std::is_same_v<Derived<T, N>, Homogeneous<T, N>>) {
38 return std::format("Homogeneous<{}, {}>", typeid(T).name(), N);
39 } else {
40 return "Unknown";
41 }
42}

48template <template <typename, int> typename Derived, typename T, int N>
49std::string to_string(const Tuple<Derived, T, N>& t) {
50 std::ostringstream oss;
51 oss << std::fixed << std::setprecision(6);
