195 auto ray_object = m_render_to_object.transform_ray(ray);
196 const Shape<ShapeType<T>, T>& shape_iface = m_shape;
197 return shape_iface.is_intersected(ray_object);
198 }

200 std::optional<std::pair<geometry::SurfaceInteraction<T>, T>> intersect_impl(const geometry::Ray<T, 3>& ray) const {
201 auto ray_object = m_render_to_object.transform_ray(ray);
202
203 const Shape<ShapeType<T>, T>& shape_iface = m_shape;
204 auto result = shape_iface.intersect(ray_object);
205 if (!result.has_value())
206 return std::nullopt;
207
208 auto [si_object, t_hit] = result.value();
209 auto si_render = m_object_to_render.transform_surface_interaction(si_object);
210 return std::make_optional(std::make_pair(si_render, t_hit));
211 }

218 const math::Point<T, 2>& u_sample

221 auto sample_opt = shape_iface.sample_on_shape(u_sample);
222 if (!sample_opt.has_value())

225 auto sample_object = sample_opt.value();

229 return std::make_optional(ShapeSample<T>{
230 point_render,

232 sample_object.uv,
