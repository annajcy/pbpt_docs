28class UniformDistribution1D : public Distribution<1> {

31
32public:
33 UniformDistribution1D(const Pt1& a, const Pt1& b) : m_a(a), m_b(b) {
34 if (a.x() >= b.x()) {
35 throw std::invalid_argument("Lower bound must be less than upper bound");
36 }
37 }
38 ~UniformDistribution1D() = default;
39
40 Point<Float, 1> sample(const Point<Float, 1>& u) const override {
41 if (u[0] < 0.0 || u[0] > 1.0) {
42 throw std::invalid_argument("Input u must be in [0,1]");
43 }
44 return m_a + u[0] * (m_b - m_a);

51 return 1.0 / (m_b.x() - m_a.x());
52 }

55 const Pt1& upper_bound() const { return m_b; }

60private:

63public:
64 UniformDistributionND(const Point<Float, N>& min_pt, const Point<Float, N>& max_pt) : m_min(min_pt), m_max(max_pt) {
65 for (int i = 0; i < N; ++i) {
66 if (min_pt[i] >= max_pt[i]) {
67 throw std::invalid_argument(
68 "Lower bounds must be less than "
69 "upper bounds in all dimensions");
70 }
71 }
72 }
73
74 Point<Float, N> sample(const Point<Float, N>& u) const override {
75 for (int i = 0; i < N; ++i) {
76 if (u[i] < 0.0 || u[i] > 1.0) {
77 throw std::invalid_argument("Input u must be in [0,1]^N");
78 }
79 }
80
81 Point<Float, N> result;
82 for (int i = 0; i < N; ++i) {
83 result[i] = m_min[i] + u[i] * (m_max[i] - m_min[i]);
84 }
85 return result;
86 }
87
88 Float pdf(const Point<Float, N>& p) const override {
89 for (int i = 0; i < N; ++i) {
90 if (p[i] < m_min[i] || p[i] > m_max[i]) {
91 return 0.0;
92 }
93 }
94
95 Float volume = 1.0;
96 for (int i = 0; i < N; ++i) {
97 volume *= (m_max[i] - m_min[i]);
98 }
99 return 1.0 / volume;
100 }
101
102 const Point<Float, N>& lower_bounds() const { return m_min; }
103 const Point<Float, N>& upper_bounds() const { return m_max; }
104};

117 Point<Float, 1> sample(const Point<Float, 1>& u) const override {
118 if (u[0] < 0.0 || u[0] > 1.0) {
119 throw std::invalid_argument("Input u must be in [0,1]");
120 }
121
122 // Use Box-Muller transform (simplified version)
123 // For a more robust implementation, consider using inverse CDF
124 Float z = std::sqrt(-2.0 * std::log(u[0])) * std::cos(2.0 * M_PI * 0.5);
125 return Point<Float, 1>{m_mean + m_stddev * z};
126 }
127
128 Float pdf(const Point<Float, 1>& p) const override {
129 Float x = p.x();
130 Float exponent = -0.5 * std::pow((x - m_mean) / m_stddev, 2);
131 return std::exp(exponent) / (m_stddev * std::sqrt(2.0 * M_PI));
132 }
