45 static std::string format_duration(double seconds) {
46 const double clamped_seconds = seconds < 0.0 ? 0.0 : seconds;
47 const auto total_seconds = static_cast<long long>(clamped_seconds + 0.5);
48 const auto hours = total_seconds / 3600;
49 const auto minutes = (total_seconds % 3600) / 60;
50 const auto secs = total_seconds % 60;
51
52 std::ostringstream oss;
53 oss << std::setfill('0') << std::setw(2) << hours << ':'
54 << std::setfill('0') << std::setw(2) << minutes << ':'
55 << std::setfill('0') << std::setw(2) << secs;
56 return oss.str();
57 }

60 static std::string format_rate(std::size_t steps, double seconds) {
61 if (steps == 0 || seconds <= 0.0) {
62 return "?it/s";
63 }
64 std::ostringstream oss;
65 oss << std::fixed << std::setprecision(2) << (steps / seconds) << "it/s";
66 return oss.str();

70 std::ostream &render(std::ostream &os) {
71 const auto current_step = std::min(m_completed_steps, m_total_steps);
72 const double clamped = (
73 m_total_steps == 0
74 ? 1.0
75 : std::clamp(
76 static_cast<double>(current_step) / static_cast<double>(m_total_steps),
77 0.0,
78 1.0
79 )
80 );
81 const auto now = std::chrono::steady_clock::now();
82 const double elapsed_seconds = std::chrono::duration<double>(now - m_start_time).count();
83 const int filled = static_cast<int>(clamped * static_cast<double>(m_bar_width));
84
85 os << '\r';
86 if (!m_description.empty()) {
87 os << m_description << ' ';
88 }
89 os << std::setw(3) << static_cast<int>(clamped * 100.0 + 0.5) << "%|";
90 for (int i = 0; i < m_bar_width; ++i) {
91 os << (i < filled ? '=' : ' ');
92 }
93 os << "| " << current_step << "/" << m_total_steps << " ";
94
95 const auto eta_seconds = clamped > 0.0
96 ? elapsed_seconds * (1.0 - clamped) / clamped
97 : -1.0;
98 os << "[" << format_duration(elapsed_seconds) << "<"
99 << (eta_seconds >= 0.0 ? format_duration(eta_seconds) : "??:??:??")
100 << ", " << format_rate(current_step, elapsed_seconds) << "]";
101 os.flush();
102 return os;
103 }
