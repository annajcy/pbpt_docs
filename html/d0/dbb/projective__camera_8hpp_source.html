192 math::Vector<int, 2> film_resolution_impl() const {
193 const auto viewport = m_projection.clip_to_viewport();
194 const auto& mat = viewport.matrix();
195 auto width_value = mat.at(0, 0) * T(2);
196 auto height_value = mat.at(1, 1) * T(2);
