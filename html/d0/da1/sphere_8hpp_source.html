117 std::pair<std::optional<T>, std::optional<T>> basic_intersect(const geometry::Ray<T, 3>& ray) const {
118 auto a = ray.direction().length_squared();
119 auto origin_vec = ray.origin().to_vector();
120 auto b = static_cast<T>(2) * ray.direction().dot(origin_vec);
121 auto c = origin_vec.length_squared() - m_radius * m_radius;
122
123 auto discriminant = b * b - static_cast<T>(4) * a * c;
124 if (discriminant < static_cast<T>(0)) {
125 return {std::nullopt, std::nullopt};
126 }
127
128 auto sqrt_discriminant = std::sqrt(discriminant);
129 auto t0 = (-b - sqrt_discriminant) / (static_cast<T>(2) * a);
130 auto t1 = (-b + sqrt_discriminant) / (static_cast<T>(2) * a);
131 return {t0, t1};
132 }

134 std::optional<IntersectionResult<T>> is_intersected(const geometry::Ray<T, 3>& ray) const {
135 auto [t0_opt, t1_opt] = basic_intersect(ray);
136 if (!t0_opt.has_value()) {
137 return std::nullopt;
138 }
139
140 auto t0 = t0_opt.value();
141 auto t1 = t1_opt.value();
142
143 if (t0 > ray.t_max() || t1 < ray.t_min()) {
144 return std::nullopt;
145 }
146
147 auto t_hit = t0;
148 if (t_hit < ray.t_min()) {
149 t_hit = t1;
150 if (t_hit > ray.t_max()) {
151 return std::nullopt;
152 }
153 }
154
155 auto p_hit = ray.at(t_hit);
156 auto phi = std::atan2(p_hit.y(), p_hit.x());
157 if (phi < static_cast<T>(0)) {
158 phi += static_cast<T>(2 * math::pi_v<T>);
159 }
160
161 if (p_hit.z() < m_z_min || p_hit.z() > m_z_max || phi > m_phi_max) {
162 return std::nullopt;
163 }
164
165 return std::make_optional(IntersectionResult<T>{
166 p_hit, t_hit, phi
167 });

190 geometry::SurfaceInteraction<T> get_surface_interaction(
191 const geometry::Ray<T, 3>& ray,
192 const IntersectionResult<T>& intersection
193 ) const {
194 T u = intersection.phi / m_phi_max;
195 T cos_theta = intersection.p_hit.z() / m_radius;
196 T theta = std::acos(std::clamp(cos_theta, static_cast<T>(-1), static_cast<T>(1)));
197 T v = (theta - z_min_theta()) / (z_max_theta() - z_min_theta());
198
199 // Compute sphere dpdu and dpdv
200 math::Vector<T, 3> dpdu(
201 -m_phi_max * intersection.p_hit.y(),
202 m_phi_max * intersection.p_hit.x(),
203 static_cast<T>(0)
204 );
205 T sin_theta = std::sqrt(1.0 - cos_theta * cos_theta);
206 T z_radius = std::sqrt(
207 intersection.p_hit.x() * intersection.p_hit.x() +
208 intersection.p_hit.y() * intersection.p_hit.y()
209 );
210 T cos_phi{}, sin_phi{};
211 bool is_polar{};
212 if (std::abs(z_radius) <= epsilon_v<T>) {
213 cos_phi = 1;
214 sin_phi = 0;
215 is_polar = true;
216 } else {
217 cos_phi = intersection.p_hit.x() / z_radius;
218 sin_phi = intersection.p_hit.y() / z_radius;
219 is_polar = false;
220 }
221
222 math::Vector<T, 3> dpdv(
223 intersection.p_hit.z() * cos_phi,
224 intersection.p_hit.z() * sin_phi,
225 -m_radius * sin_theta
226 );
227 dpdv = dpdv * (z_max_theta() - z_min_theta());
228
229 // Compute sphere d2pduu, d2pdudv, d2pdvv
230 auto d2pduu = -m_phi_max * m_phi_max * math::Vector<T, 3>(
231 intersection.p_hit.x(),
232 intersection.p_hit.y(),
233 static_cast<T>(0)
234 );
235
236 auto d2pdudv = (z_max_theta() - z_min_theta()) * intersection.p_hit.z() * m_phi_max * math::Vector<T, 3>(
237 -sin_phi,
238 cos_phi,
239 static_cast<T>(0)
240 );
241 auto d2pdvv = -std::pow(z_max_theta() - z_min_theta(), 2) * math::Vector<T, 3>(
242 intersection.p_hit.x(),
243 intersection.p_hit.y(),
244 intersection.p_hit.z()
245 );
246
247 math::Vector<T, 3> N{};
248
249 // Compute fundamental forms
250 T E = dpdu.dot(dpdu);
251 T F = dpdu.dot(dpdv);
252 T G = dpdv.dot(dpdv);
253 if (!is_polar) {
254 N = cross(dpdu, dpdv).normalized();
255 } else {
256 auto n = (intersection.p_hit - math::Point<T, 3>(0, 0, 0));
257 N = n.normalized();
258 }
259
260 T e = N.dot(d2pduu);
261 T f = N.dot(d2pdudv);
262 T g = N.dot(d2pdvv);
263
264 T EGF2 = E * G - F * F;
265 T inv_EGF2 = (EGF2 == static_cast<T>(0)) ? static_cast<T>(0) : static_cast<T>(1) / EGF2;
266 math::Vector<T, 3> dndu = ((f * F - e * G) * inv_EGF2) * dpdu + ((e * F - f * E) * inv_EGF2) * dpdv;
267 math::Vector<T, 3> dndv = ((g * F - f * G) * inv_EGF2) * dpdu + ((f * F - g * E) * inv_EGF2) * dpdv;
268
269 math::Vector<T, 3> p_error = math::gamma<T>(5) * math::Vector<T, 3>(
270 std::abs(intersection.p_hit.x()),
271 std::abs(intersection.p_hit.y()),
272 std::abs(intersection.p_hit.z())
273 );
274
275 return geometry::SurfaceInteraction<T>(
276 intersection.p_hit - p_error,
277 intersection.p_hit + p_error,
278 -ray.direction().normalized(),
279 math::Normal<T, 3>(N),
280 math::Point<T, 2>(u, v),
281 dpdu,
282 dpdv,
283 math::Normal<T, 3>(dndu),
284 math::Normal<T, 3>(dndv)
285 );
286 }

299 return std::nullopt;

301 math::Vector<T, 3> p_error = math::gamma<T>(5) * math::Vector<T, 3>(
302 std::abs(p.x()),
303 std::abs(p.y()),
304 std::abs(p.z())
305 );
306
307 math::Normal<T, 3> n = math::Normal<T, 3>::from_vector(
308 (p - math::Point<T, 3>(0, 0, 0)).normalized()
309 );
310
311 T theta = std::acos(std::clamp(p.z() / m_radius, static_cast<T>(-1), static_cast<T>(1)));
312 T phi = std::atan2(p.y(), p.x());
313 if (phi < static_cast<T>(0)) {

315 }
316
317 if (phi > m_phi_max) {

320
321 T u = phi / m_phi_max;
322 T v = (theta - z_min_theta()) / (z_max_theta() - z_min_theta());

324 sample.point = p;
325 sample.normal = n;
326 sample.uv = math::Point<T, 2>(u, v);
