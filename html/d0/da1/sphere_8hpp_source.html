121 std::pair<std::optional<T>, std::optional<T>> basic_intersect(const geometry::Ray<T, 3>& ray) const {
122 auto a = ray.direction().length_squared();
123 auto origin_vec = ray.origin().to_vector();
124 auto b = static_cast<T>(2) * ray.direction().dot(origin_vec);
125 auto c = origin_vec.length_squared() - m_radius * m_radius;
126
127 auto discriminant = b * b - static_cast<T>(4) * a * c;
128 if (discriminant < static_cast<T>(0)) {
129 return {std::nullopt, std::nullopt};
130 }
131
132 auto sqrt_discriminant = std::sqrt(discriminant);
133 auto t0 = (-b - sqrt_discriminant) / (static_cast<T>(2) * a);
134 auto t1 = (-b + sqrt_discriminant) / (static_cast<T>(2) * a);
135 return {t0, t1};
136 }

138 std::optional<IntersectionResult<T>> is_intersected(const geometry::Ray<T, 3>& ray) const {
139 auto [t0_opt, t1_opt] = basic_intersect(ray);
140 if (!t0_opt.has_value()) {
141 return std::nullopt;
142 }
143
144 auto t0 = t0_opt.value();
145 auto t1 = t1_opt.value();
146
147 if (t0 > ray.t_max() || t1 < ray.t_min()) {
148 return std::nullopt;
149 }
150
151 auto t_hit = t0;
152 if (t_hit < ray.t_min()) {
153 t_hit = t1;
154 if (t_hit > ray.t_max()) {
155 return std::nullopt;
156 }
157 }
158
159 auto p_hit = ray.at(t_hit);
160 auto phi = std::atan2(p_hit.y(), p_hit.x());
161 if (phi < static_cast<T>(0)) {
162 phi += static_cast<T>(2 * math::pi_v<T>);
163 }
164
165 if (p_hit.z() < m_z_min || p_hit.z() > m_z_max || phi > m_phi_max) {
166 return std::nullopt;
167 }
168
169 return std::make_optional(IntersectionResult<T>{
170 p_hit, t_hit, phi
171 });

194 geometry::SurfaceInteraction<T> get_surface_interaction(
195 const geometry::Ray<T, 3>& ray,
196 const IntersectionResult<T>& intersection
197 ) const {
198 T u = intersection.phi / m_phi_max;
199 T cos_theta = intersection.p_hit.z() / m_radius;
200 T theta = std::acos(std::clamp(cos_theta, static_cast<T>(-1), static_cast<T>(1)));
201 T v = (theta - z_min_theta()) / (z_max_theta() - z_min_theta());
202
203 // Compute sphere dpdu and dpdv
204 math::Vector<T, 3> dpdu(
205 -m_phi_max * intersection.p_hit.y(),
206 m_phi_max * intersection.p_hit.x(),
207 static_cast<T>(0)
208 );
209 T sin_theta = std::sqrt(1.0 - cos_theta * cos_theta);
210 T z_radius = std::sqrt(
211 intersection.p_hit.x() * intersection.p_hit.x() +
212 intersection.p_hit.y() * intersection.p_hit.y()
213 );
214 T cos_phi{}, sin_phi{};
215 bool is_polar{};
216 if (std::abs(z_radius) <= epsilon_v<T>) {
217 cos_phi = 1;
218 sin_phi = 0;
219 is_polar = true;
220 } else {
221 cos_phi = intersection.p_hit.x() / z_radius;
222 sin_phi = intersection.p_hit.y() / z_radius;
223 is_polar = false;
224 }
225
226 math::Vector<T, 3> dpdv(
227 intersection.p_hit.z() * cos_phi,
228 intersection.p_hit.z() * sin_phi,
229 -m_radius * sin_theta
230 );
231 dpdv = dpdv * (z_max_theta() - z_min_theta());
232
233 // Compute sphere d2pduu, d2pdudv, d2pdvv
234 auto d2pduu = -m_phi_max * m_phi_max * math::Vector<T, 3>(
235 intersection.p_hit.x(),
236 intersection.p_hit.y(),
237 static_cast<T>(0)
238 );
239
240 auto d2pdudv = (z_max_theta() - z_min_theta()) * intersection.p_hit.z() * m_phi_max * math::Vector<T, 3>(
241 -sin_phi,
242 cos_phi,
243 static_cast<T>(0)
244 );
245 auto d2pdvv = -std::pow(z_max_theta() - z_min_theta(), 2) * math::Vector<T, 3>(
246 intersection.p_hit.x(),
247 intersection.p_hit.y(),
248 intersection.p_hit.z()
249 );
250
251 math::Vector<T, 3> N{};
252
253 // Compute fundamental forms
254 T E = dpdu.dot(dpdu);
255 T F = dpdu.dot(dpdv);
256 T G = dpdv.dot(dpdv);
257 if (!is_polar) {
258 N = cross(dpdu, dpdv).normalized();
259 } else {
260 auto n = (intersection.p_hit - math::Point<T, 3>(0, 0, 0));
261 N = n.normalized();
262 }
263
264 T e = N.dot(d2pduu);
265 T f = N.dot(d2pdudv);
266 T g = N.dot(d2pdvv);
267
268 T EGF2 = E * G - F * F;
269 T inv_EGF2 = (EGF2 == static_cast<T>(0)) ? static_cast<T>(0) : static_cast<T>(1) / EGF2;
270 math::Vector<T, 3> dndu = ((f * F - e * G) * inv_EGF2) * dpdu + ((e * F - f * E) * inv_EGF2) * dpdv;
271 math::Vector<T, 3> dndv = ((g * F - f * G) * inv_EGF2) * dpdu + ((f * F - g * E) * inv_EGF2) * dpdv;
272
273 math::Vector<T, 3> p_error = math::gamma<T>(5) * math::Vector<T, 3>(
274 std::abs(intersection.p_hit.x()),
275 std::abs(intersection.p_hit.y()),
276 std::abs(intersection.p_hit.z())
277 );
278
279 return geometry::SurfaceInteraction<T>(
280 intersection.p_hit - p_error,
281 intersection.p_hit + p_error,
282 -ray.direction().normalized(),
283 math::Normal<T, 3>(N),
284 math::Point<T, 2>(u, v),
285 dpdu,
286 dpdv,
287 math::Normal<T, 3>(dndu),
288 math::Normal<T, 3>(dndv)
289 );
290 }
