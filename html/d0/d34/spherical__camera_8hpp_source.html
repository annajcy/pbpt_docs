81 geometry::Ray<T, 3> generate_ray_impl(const CameraSample<T>& sample) const {
82 math::Point<T, 2> uv = math::Point<T, 2>(
83 sample.p_film.x() / static_cast<T>(m_film_resolution.x()),
84 sample.p_film.y() / static_cast<T>(m_film_resolution.y())
85 );
86 T theta = math::pi_v<T> * uv.x(), phi = 2 * math::pi_v<T> * uv.y();
87
88 math::Vector<T, 3> dir;
89 if (m_mapping == SphericalCameraMapping::EqualRectangular) {
90 dir = geometry::SphericalPoint<T, 3>(math::Vector<T, 2>{theta, phi}, 1).to_cartesian();
91 } else if (m_mapping == SphericalCameraMapping::EqualArea) {
92 dir = geometry::equal_area_square_to_sphere(
93 geometry::warp_equal_area_square(uv).to_vector()
94 );
95 }
96 std::swap(dir.y(), dir.z());
97 return geometry::Ray<T, 3>(math::Point<T, 3>(0, 0, 0), dir);
98 }

105 auto p_film_x = sample.p_film + math::Vector<T, 2>(eps, 0);
